public static boolean pruefeVertrag(
        Vertrag vertrag ,
        // Zentrale oder Regionalmanagement
        ORB_Z_R_OREConstraint constraint ) {

   // anonyme innere Klasse des übergebenen Constraint
   return constraint.new ORB_Z_R_OREConstraint() {
        @Override
        protected Boolean caseZ(
                // Zentrale
                ZConstraint zConstraint ) {
            return pruefeVertrag(
                    vertrag ,
                    zConstraint );
        }

        @Override
        protected Boolean caseR(
                // Regionalmanagement
                RConstraint rConstraint ) {
            ...
        }
    }.doSwitch();
}
